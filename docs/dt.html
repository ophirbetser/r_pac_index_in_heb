dt_ <- function(df){
  DT::datatable(
  df, 
  extensions = c('Buttons', 'ColReorder', 'Scroller'),
  options = list(
    dom = 'Brtip',
    buttons = c('copy', 'csv', 'pdf', 'colvis'),
    colReorder = list(realtime = FALSE),
    scroller = F
  )
)
}
